function getElement(stem){
  return elementMap[stem] || "";
}

function getYearPillar(y, m, d){
  let adjustedYear = y;
  if(m < 2 || (m === 2 && d < 4)) adjustedYear -= 1;
  const i = adjustedYear - 1984;
  return stems[(i % 10 + 10) % 10] + branches[(i % 12 + 12) % 12];
}

function getMonthBranch(m, d){
  const approx = [
    {m:2,d:4,b:"寅"},{m:3,d:6,b:"卯"},{m:4,d:5,b:"辰"},{m:5,d:6,b:"巳"},
    {m:6,d:6,b:"午"},{m:7,d:7,b:"未"},{m:8,d:8,b:"申"},{m:9,d:8,b:"酉"},
    {m:10,d:8,b:"戌"},{m:11,d:7,b:"亥"},{m:12,d:7,b:"子"},{m:1,d:6,b:"丑"}
  ];
  let b = "丑";
  for(const item of approx){
    if(m > item.m || (m === item.m && d >= item.d)) b = item.b;
  }
  return b;
}

function getMonthPillar(yearStem, m, d){
  const startMap = {
    "甲":"丙","己":"丙",
    "乙":"戊","庚":"戊",
    "丙":"庚","辛":"庚",
    "丁":"壬","壬":"壬",
    "戊":"甲","癸":"甲"
  };
  const order = {"寅":0,"卯":1,"辰":2,"巳":3,"午":4,"未":5,"申":6,"酉":7,"戌":8,"亥":9,"子":10,"丑":11};
  const monthBranch = getMonthBranch(m, d);
  const startStem = startMap[yearStem];
  const startIndex = stems.indexOf(startStem);
  return stems[(startIndex + order[monthBranch]) % 10] + monthBranch;
}

function getJulianDay(y, m, d){
  let yy = y;
  let mm = m;
  if(mm <= 2){
    yy -= 1;
    mm += 12;
  }
  const A = Math.floor(yy / 100);
  const B = 2 - A + Math.floor(A / 4);
  return Math.floor(365.25 * (yy + 4716)) + Math.floor(30.6001 * (mm + 1)) + d + B - 1524;
}

function getDayPillar(y, m, d){
  const jd = getJulianDay(y, m, d);
  const index = jd - 11;
  return stems[(index % 10 + 10) % 10] + branches[(index % 12 + 12) % 12];
}

function getHourBranch(hour){
  const idx = Math.floor(((hour + 1) % 24) / 2);
  return branches[idx];
}

function getHourPillar(dayStem, hour){
  const startMap = {

  const hb = getHourBranch(hour);
  const startStem = startMap[dayStem];
  const idx = stems.indexOf(startStem);
  return stems[(idx + order[hb]) % 10] + hb;
}

function produces(a, b){
  return (
    (a==="木"&&b==="火") ||
    (a==="火"&&b==="土") ||
    (a==="土"&&b==="金") ||
    (a==="金"&&b==="水") ||
    (a==="水"&&b==="木")
  );
}

function controls(a, b){
  return (
    (a==="木"&&b==="土") ||
    (a==="土"&&b==="水") ||
    (a==="水"&&b==="火") ||
    (a==="火"&&b==="金") ||

  );
}

function getTsuhen(dayStem, targetStem){
  const dayElement = elementMap[dayStem];
  const targetElement = elementMap[targetStem];
  const samePolarity = yinYangMap[dayStem] === yinYangMap[targetStem];

  if(dayElement === targetElement) return samePolarity ? "比肩" : "劫財";
  if(produces(dayElement, targetElement)) return samePolarity ? "食神" : "傷官";
  if(controls(dayElement, targetElement)) return samePolarity ? "偏財" : "正財";
  if(controls(targetElement, dayElement)) return samePolarity ? "偏官" : "正官";
  if(produces(targetElement, dayElement)) return samePolarity ? "偏印" : "印綬";
  return "";
}

function getJuniun(dayStem, branch){
  const list = twelveFortuneTable[dayStem] || [];
  const found = list.find(v => v.startsWith(branch + ":"));
  return found ? found.split(":")[1] : "";
}

function getLuckDirection(yearStem, gender){
  const isYang = ["甲","丙","戊","庚","壬"].includes(yearStem);
  if(gender === "男性") return isYang ? "順行" : "逆行";
  if(gender === "女性") return isYang ? "逆行" : "順行";
  return "順行";
}

function getDaiun(monthStem, monthBranch, yearStem, gender){
  let direction = getLuckDirection(yearStem, gender);
  let stemIndex = stems.indexOf(monthStem);
  let branchIndex = branches.indexOf(monthBranch);
  let arr = [];
  let startAge = 8;

  for(let i=0;i<8;i++){
    if(direction === "順行"){
      stemIndex = (stemIndex + 1) % 10;
      branchIndex = (branchIndex + 1) % 12;
    }else{
      stemIndex = (stemIndex - 1 + 10) % 10;
      branchIndex = (branchIndex - 1 + 12) % 12;
    }
    arr.push(`${startAge + i*10}歳〜${startAge + i*10 + 9}歳：${stems[stemIndex]}${branches[branchIndex]}`);
  }
  return arr;
}

function buildPerson(birth, birthTime, gender){
  const [y,m,d] = birth.split("-").map(Number);
  const hour = birthTime ? Number(birthTime.split(":")[0]) : 12;

  const dayP = getDayPillar(y,m,d);
  const hourP = getHourPillar(dayP[0],hour);

  return {
    y,m,d,hour,gender,
    yearP, monthP, dayP, hourP,
    dayStem: dayP[0],
    element: getElement(dayP[0]),
    hidden:{

      month:getHiddenStems(monthP[1]),
      day:getHiddenStems(dayP[1]),
      hour:getHiddenStems(hourP[1])
    }
  };
}

function getHexLine(){
  const toss = () => (Math.random() < 0.5 ? 2 : 3);
  const total = toss() + toss() + toss();
  if(total === 6) return {value:0, changing:true, label:"老陰"};
  if(total === 7) return {value:1, changing:false, label:"少陽"};
  if(total === 8) return {value:0, changing:false, label:"少陰"};
  return {value:1, changing:true, label:"老陽"};
}

function getTrigramName(bits){
  const map = {
    "111":"乾","110":"兌","101":"離","100":"震",
    "011":"巽","010":"坎","001":"艮","000":"坤"
  };
  return map[bits] || "";
}

function renderYao(line){

}

function buildIChing(){
  const lines = [];
  for(let i=0;i<6;i++) lines.push(getHexLine());

  const originalBits = lines.map(l => l.value);
  const changedBits = lines.map(l => l.changing ? (l.value === 1 ? 0 : 1) : l.value);

  const lower = originalBits.slice(0,3).reverse().join("");
  const upper = originalBits.slice(3,6).reverse().join("");
  const lowerChanged = changedBits.slice(0,3).reverse().join("");
  const upperChanged = changedBits.slice(3,6).reverse().join("");

  const originalName = `${getTrigramName(upper)}上${getTrigramName(lower)}下`;
  const changedName = `${getTrigramName(upperChanged)}上${getTrigramName(lowerChanged)}下`;

  const changingLines = lines.map((l,i) => l.changing ? `${i+1}爻` : null).filter(Boolean);

  return { lines, originalName, changedName, changingLines };
}

function getCompatibility(person, partner, partnerName){
  const dayElement1 = person.element;
  const dayElement2 = partner.element;
  const dayStem1 = person.dayStem;
  const dayStem2 = partner.dayStem;

  let score = 0;
  let text = [];

  if(dayElement1 === dayElement2){
    score += 40;
    text.push("同じ五行を持つため、根本的な感覚が通じやすい相性です。");
  } else {
    text.push("異なる五行を持つため、違いを理解することで補い合いやすい相性です。");
  }

  const goodPairs = [["木","火"],["火","土"],["土","金"],["金","水"],["水","木"]];
  if(goodPairs.some(pair => pair[0] === dayElement1 && pair[1] === dayElement2) ||
     goodPairs.some(pair => pair[0] === dayElement2 && pair[1] === dayElement1)){
    score += 30;
    text.push("五行の流れがつながりやすく、関係を育てやすい相です。");
  }

  if(dayStem1 === dayStem2){
    score += 20;
    text.push("日主が同じため、考え方や心の動きが似やすい関係です。");
  }

  let label = "努力で深まる相性";
  if(score >= 70) label = "非常に相性が良い";
  else if(score >= 40) label = "学び合える良い相性";

  return `相手：${partnerName}
相手の日柱：${partner.dayP}
相手の日主：${partner.dayStem}（${partner.element}）
判定：${label}

${text.join("\n")}`;
}

function uranai(){
  const name = document.getElementById("name").value.trim();
  const birth = document.getElementById("birth").value;
  const birthTime = document.getElementById("birthTime").value;
  const gender = document.getElementById("gender").value;
  const partnerName = document.getElementById("partnerName").value.trim();
  const partnerBirth = document.getElementById("partnerBirth").value;
  const partnerBirthTime = document.getElementById("partnerBirthTime").value;
  const question = document.getElementById("question").value.trim();

  if(!name || !birth){
    alert("お名前と生年月日を入力してください");
    return;
  }

  const person = buildPerson(birth, birthTime, gender);
  const daiun = getDaiun(person.monthP[0], person.monthP[1], person.yearP[0], gender);
  const nenun = getYearPillarFull(new Date().getFullYear());
  const monthly = getMonthlyFortunes();
  const eki = buildIChing();
  const categoryText = getCategoryAdvice(question, person.element);

  let compatibilityText = "お相手の情報が未入力です。";
  if(partnerName && partnerBirth){
    const partner = buildPerson(partnerBirth, partnerBirthTime, "その他");
    compatibilityText = getCompatibility(person, partner, partnerName);
  }

  const report = buildReport(name, person, question, categoryText, daiun, nenun, monthly, compatibilityText, eki);

  document.getElementById("plainResult").innerText = report.fullText;
  document.getElementById("reportMeishiki").innerText = report.meishikiText;
  document.getElementById("reportFlow").innerText = report.flowText;
  document.getElementById("reportAdvice").innerText = report.adviceText;
  document.getElementById("reportView").classList.remove("hidden");
}

function copyText(){
  const text = document.getElementById("plainResult").innerText;
  if(!text){
    alert("先に鑑定してください");
    return;
  }
  navigator.clipboard.writeText(text);
  alert("コピーしました");
}

function saveHistory(){
  const text = document.getElementById("plainResult").innerText;
  if(!text){
    alert("先に鑑定してください");
    return;
  }
  const history = JSON.parse(localStorage.getItem("fortune_history") || "[]");
  history.unshift({
    createdAt: new Date().toLocaleString("ja-JP"),
    text
  });
  localStorage.setItem("fortune_history", JSON.stringify(history.slice(0, 20)));
  alert("履歴を保存しました");
}

function loadLatestHistory(){
  const history = JSON.parse(localStorage.getItem("fortune_history") || "[]");
  if(history.length === 0){
    alert("履歴がありません");
    return;
  }
  document.getElementById("plainResult").innerText = history[0].text;
  document.getElementById("reportView").classList.remove("hidden");
  document.getElementById("reportMeishiki").innerText = history[0].text;
  document.getElementById("reportFlow").innerText = "";
  document.getElementById("reportAdvice").innerText = "";
}
